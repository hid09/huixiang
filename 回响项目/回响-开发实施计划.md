#### 1.4 数据库Schema设计
```sql
-- 用户表（单人版）
CREATE TABLE users (
    id TEXT PRIMARY KEY,
    name TEXT DEFAULT '回响用户',
    avatar TEXT,
    timezone TEXT DEFAULT 'Asia/Shanghai',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 语音记录表
CREATE TABLE voice_records (
    id TEXT PRIMARY KEY,
    user_id TEXT REFERENCES users(id),
    audio_url TEXT,              -- 语音文件URL
    audio_duration INTEGER,      -- 时长（秒）
    transcribed_text TEXT,       -- 转写文本
    primary_emotion TEXT,        -- 主导情绪
    emotion_intensity INTEGER,   -- 情绪强度0-10
    topic_tags TEXT,             -- 话题标签JSON
    recorded_at TIMESTAMP,       -- 记录时间
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 每日日记表
CREATE TABLE daily_diaries (
    id TEXT PRIMARY KEY,
    user_id TEXT REFERENCES users(id),
    diary_date DATE UNIQUE,
    emotion_summary TEXT,
    events_summary TEXT,
    thoughts_summary TEXT,
    small_discovery TEXT,
    ai_tone TEXT,
    emotion_journey TEXT,
    closing_message TEXT,
    cognitive_change TEXT,       -- JSON格式
    generated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 复盘报告表
CREATE TABLE review_reports (
    id TEXT PRIMARY KEY,
    user_id TEXT REFERENCES users(id),
    report_type TEXT,            -- weekly/monthly/quarterly/yearly
    period_start DATE,
    period_end DATE,
    content TEXT,                -- JSON格式
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 认知变化记录表
CREATE TABLE cognitive_changes (
    id TEXT PRIMARY KEY,
    user_id TEXT REFERENCES users(id),
    topic TEXT,
    past_view TEXT,
    past_date DATE,
    past_emotion TEXT,
    current_view TEXT,
    current_date DATE,
    current_emotion TEXT,
    insight TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 统计表
CREATE TABLE user_stats (
    user_id TEXT PRIMARY KEY REFERENCES users(id),
    total_record_days INTEGER DEFAULT 0,
    total_voice_count INTEGER DEFAULT 0,
    continuous_days INTEGER DEFAULT 0,
    longest_continuous_days INTEGER DEFAULT 0,
    last_record_date DATE,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

  ```python
  # 降级链示例
  MODEL_FALLBACK_CHAIN = {
      'diary_generation': ['qwen-plus', 'qwen-turbo'],
      'monthly_review': ['qwen-max', 'qwen-plus', 'kimi'],
      'cognitive_change': ['kimi', 'qwen-max'],
  }

  async def call_with_fallback(task_type, prompt):
      models = MODEL_FALLBACK_CHAIN[task_type]
      for model in models:
          try:
              return await call_model(model, prompt, timeout=30)
          except (TimeoutError, APIError) as e:
              log_fallback(model, e)
              continue
      raise AllModelsFailedError(task_type)
  ```
